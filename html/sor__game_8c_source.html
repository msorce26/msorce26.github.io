178  switch (c) {

189  snake_game.dir = 2;//dir flag set to down
190  break;
191  case 'd':
192  if(snake_game.dir != 3)//if dir flag not set to left
193  snake_game.dir = 1;//dir flag set to right
194  break;
195  default:
196  break;
197  }
